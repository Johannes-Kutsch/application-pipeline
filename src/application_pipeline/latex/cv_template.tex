% cv_template.tex — single source of truth for cover letter + resume.
% Three outputs via the \BUILD flag:
%   cover    → cover letter only (1 page)
%   resume   → resume only       (2 pages)
%   combined → both, separated   (3 pages, default)
%
% User-specific data (identity, contact, content pool, photo, signature) is
% read from \CvDataDir/, which defaults to ../user-info/cv so this
% template compiles standalone. A build orchestrator overrides via
%   pdflatex "\def\CvDataDir{<abs path>}\def\BUILD{combined}\input{cv_template}"
%
% See templates/README.md for the placeholder convention and content-pool format.

\documentclass[11pt,a4paper,sans]{moderncv}

% Require moderncv >= 2.0.0 (TeX Live 2020+ or MiKTeX rolling).
\@ifclasslater{moderncv}{2015/01/01}{}{%
  \PackageError{cv_template}{moderncv >= 2.0.0 required.
  Install a current TeX distribution (TeX Live 2020+ or MiKTeX rolling).}{}}

\moderncvstyle{casual}
\moderncvcolor{blue}

\usepackage[ngerman]{babel}
\usepackage[utf8x]{inputenc}
\usepackage{enumitem}
\usepackage{ragged2e}
\usepackage{setspace}
\usepackage{xpatch}
\usepackage{etoolbox}
\usepackage{graphicx}
\usepackage[scale=0.75]{geometry}

% Ported from cv_7.tex — remove the trailing dot moderncv appends to \cventry.
% Wrapped in \AtBeginDocument so the patch fires after the v2.x body style
% files finish loading (they redefine \cventry via \moderncvstyle).
\AtBeginDocument{\xpatchcmd{\cventry}{.\strut}{\strut}{}{}}

% Ported from cv_7.tex — tighter itemize spacing.
\setlist[itemize]{itemsep=-0.2em}
\setlist[itemize,2]{topsep=-0.4em}

% Build flag and user-data directory — both overridable from the command line.
\providecommand{\BUILD}{combined}
\providecommand{\CvDataDir}{../user-info/cv}

% Raw facts (\def\myFirstname etc.) come from \CvDataDir/facts.
% Moderncv API calls and display-macro derivations live here, not in facts.tex.
\input{\CvDataDir/facts}
\firstname{\myFirstname}
\familyname{\myFamilyname}
\title{Lebenslauf}
\photo[120pt][1pt]{\CvDataDir/profile}
\address{\myStreet}{\myZip}{}
\phone[mobile]{\myPhone}
\email{\myEmail}
\social[github]{\myGithub}
\social[linkedin]{\myLinkedin}
\def\addressdisplay{\myStreet, \myZip}
\def\phonedisplay{\myPhone}
\def\emaildisplay{\myEmail}
\def\githubdisplay{\myGithub}
\def\linkedindisplay{\myLinkedin}

% Override moderncv's letter-closing to inject the signature image automatically,
% so the applicant never has to print/sign/scan an Anschreiben. This couples to
% moderncv internals by design; a Tier-3 refactor of the class will absorb it.
\renewcommand*{\makeletterclosing}{%
  \vspace{1em}\@closing\\[\medskipamount]%
  \includegraphics[height=2.5em]{\CvDataDir/signature}\\%
  \@firstname~\@familyname%
}

\setstretch{1}
\begin{document}

%----------------------------------------------------------------------------------------
% COVER LETTER
%----------------------------------------------------------------------------------------
\ifdefstring{\BUILD}{resume}{}{%
\clearpage
% Four-line DIN 5008 Anschriftenfeld. Each line is its own slot; empty bodies
% are skipped via \ifstrempty so a missing recipient_name does not leave a gap.
% Company goes in arg 1 (rendered bold by moderncv's casual style); the other
% three lines are stacked in arg 2 with \\ between non-empty lines.
\recipient{<<RECIPIENT_COMPANY>>}{%
  \ifstrempty{<<RECIPIENT_NAME>>}{}{<<RECIPIENT_NAME>>\\}%
  \ifstrempty{<<RECIPIENT_STREET>>}{}{<<RECIPIENT_STREET>>\\}%
  <<RECIPIENT_ZIP_CITY>>%
}
\date{\today}
\opening{<<OPENING>>}
\closing{Mit freundlichen Grüßen,}

\makelettertitle
\setstretch{1.8}
\vspace{-1em}
\justify

<<COVER_INTRO>>

<<COVER_PIVOT>>

<<COVER_FIT>>

<<COVER_CLOSING>>

\makeletterclosing

\setstretch{1}
\ifdefstring{\BUILD}{combined}{\newpage}{}%
}

%----------------------------------------------------------------------------------------
% RESUME
%----------------------------------------------------------------------------------------
\ifdefstring{\BUILD}{cover}{}{%

% Content-pool macros (\itemFoo, \itemBar, ...) — see content_pool format in README.
\input{\CvDataDir/content_pool}

\makecvtitle
\vspace{-4.3em}

\section{Persönliche Informationen}
\cvitem{Name}{\@firstname{} \@familyname}
\cvitem{Adresse}{\addressdisplay}
\cvitem{Telefon}{\phonedisplay}
\cvitem{E-Mail}{\emaildisplay}
\cvitem{GitHub}{\githubdisplay}
\cvitem{LinkedIn}{\linkedindisplay}

\section{Berufserfahrung}
<<RESUME_BERUFSERFAHRUNG>>

\section{Ausbildung}
<<RESUME_AUSBILDUNG>>

\section{Projekte}
<<RESUME_PROJEKTE>>

\section{Kenntnisse}
<<SKILLS_BLOCK>>

\section{Sprachen}
\Languages

\section{Hobbies und Interessen}
\renewcommand{\listitemsymbol}{-~}
\Hobbies
}

\end{document}
