% cv_template.tex — single source of truth for cover letter + resume.
% Three outputs via the \BUILD flag:
%   cover    → cover letter only (1 page)
%   resume   → resume only       (2 pages)
%   combined → both, separated   (3 pages, default)
%
% User-specific data (identity, contact, content pool, photo, signature) is
% read from \CvDataDir/, which defaults to ../user-info/cv so this
% template compiles standalone. A build orchestrator overrides via
%   pdflatex "\def\CvDataDir{<abs path>}\def\BUILD{combined}\input{cv_template}"
%
% See templates/README.md for the placeholder convention and content-pool format.

\documentclass[11pt,a4paper,sans]{moderncv}

% moderncv 1.2.0 is vendored alongside this template (src/application_pipeline/latex/)
% and copied into .build/ at compile time; TEXINPUTS isolation in compile_cv_cmd.py
% hides the host's moderncv v2.x. See ADR-0034.

\moderncvstyle{casual}
\moderncvcolor{blue}

% Ported from cv_7.tex — remove the trailing dot moderncv appends to \cventry.
\xpatchcmd{\cventry}{.\strut}{\strut}{}{}

% Ported from cv_7.tex's hand-modified casual.sty — extra 2em of vertical space
% between the recipient/date row and the opening line, so the Anschriftenfeld
% and the Anrede are not visually crowded.
\makeatletter
\xpatchcmd{\makelettertitle}{\@date\\[2em]}{\@date\\[2em]\\[2em]}{}{}
\xpatchcmd{\makelettertitle}{\@opening\\[1.5em]}{\@opening\\[\CoverLetterGap]}{}{}
\makeatother

% Strip only the applicant's name from moderncv casual.sty's letter footer,
% matching the hand-modified casual.sty shipped with cv_7.tex (see
% application-pipeline/CV/moderncvstylecasual.sty:165 where the name line is
% commented out). Contact info and the fancypagestyle setup stay so page
% numbers and contact lines render normally.
\makeatletter
\xpatchcmd{\makeletterfooter}
  {\vspace{-\baselineskip}}{}{}{}
\xpatchcmd{\makeletterfooter}
  {\strut{\bfseries\upshape\@firstname~\@familyname}\\}{}{}{}
\makeatother

% Ported from cv_7.tex — tighter itemize spacing.
\setlist[itemize]{itemsep=-0.2em}
\setlist[itemize,2]{topsep=-0.4em}

% User-level helper macro on top of stock moderncv 1.2.0. Verbatim from the
% pre-#474 hand-modified moderncv.cls; lives here (not in the vendored class)
% because ADR-0034 mandates shipping moderncv 1.2.0 verbatim.
% \nopagebreak[4] keeps the sub-entry glued to its parent \cventry: moderncv
% inserts a \par between rows, so without it the optimal page-break can land
% between the title row and the Masterarbeit/Bachelorarbeit body. Items in
% content_pool.tex are intended to be atomic; splitting one across pages reads
% as a typo.
\newcommand*{\subcventry}[3]{
  \nopagebreak[4]
  \vspace{-0.25em}
  \cvline{}{\textbf{#1:} \textit{#2} \newline #3}
}

% Build flag and user-data directory — both overridable from the command line.
\providecommand{\BUILD}{combined}
\providecommand{\CvDataDir}{../user-info/cv}

% Raw facts (\def\myFirstname, \def\myFamilyname, \def\myCity, \def\PersonalInfo,
% \def\Languages, \def\Hobbies) come from \CvDataDir/facts. The template
% references those macros directly; per-row identity content lives in
% \PersonalInfo so users add/remove rows without touching this file.
\input{\CvDataDir/facts}
\firstname{\myFirstname}
\familyname{\myFamilyname}
\title{Lebenslauf}
\photo[120pt][1pt]{\CvDataDir/profile}

% Gemeinsamer Block
\newcommand{\SignatureBlock}{%
  \noindent\includegraphics[height=3em]{\CvDataDir/signature}\\
  \textbf{\myFirstname{} \myFamilyname}\\%
  \myCity{}, \today%
}

% Override moderncv's letter-closing to inject the signature image automatically,
% so the applicant never has to print/sign/scan an Anschreiben. This couples to
% moderncv internals by design; a Tier-3 refactor of the class will absorb it.
\newcommand{\CoverLetterGap}{1.75\baselineskip}
\makeatletter
\renewcommand*{\makeletterclosing}{%
  % Keep the sign-off gap proportional to the selected cover stretch so it
  % matches the opening-to-intro gap. Reset stretch to 1 only after \@closing
  % so the signature block stays tight.
  \setstretch{1}%
  \vspace{\CoverLetterGap}
  \vfill
  \@closing\par
  \SignatureBlock
}
\makeatother

\newsavebox{\CoverLetterBox}
\newdimen\CoverLetterRemaining
\newcommand{\SetCoverLetterBox}[2]{%
  \setbox\CoverLetterBox=\vbox{%
    \hsize=\textwidth
    \setstretch{#1}%
    \justify
    #2%
    \par
    \makeletterclosing
  }%
}
\newcommand{\AutoCoverLetterStretch}[5]{%
  \CoverLetterRemaining=\pagegoal
  \advance\CoverLetterRemaining by -\pagetotal
  \SetCoverLetterBox{#1}{#5}%
  \ifdim\dimexpr\ht\CoverLetterBox+\dp\CoverLetterBox\relax>\CoverLetterRemaining
    \SetCoverLetterBox{#2}{#5}%
  \fi
  \ifdim\dimexpr\ht\CoverLetterBox+\dp\CoverLetterBox\relax>\CoverLetterRemaining
    \SetCoverLetterBox{#3}{#5}%
  \fi
  \ifdim\dimexpr\ht\CoverLetterBox+\dp\CoverLetterBox\relax>\CoverLetterRemaining
    \SetCoverLetterBox{#4}{#5}%
  \fi
  \unvbox\CoverLetterBox
}
